document.getElementById("translateBtn").addEventListener("click", () => {
  const inputText = document.getElementById("inputText").value.trim();
  const fromLang = document.getElementById("fromLang").value;
  const toLang = document.getElementById("toLang").value;
  const outputText = document.getElementById("outputText");

  if (inputText === "") {
    outputText.value = "Please enter some text to translate!";
    return;
  }

  const url = `https://api.mymemory.translated.net/get?q=${encodeURIComponent(inputText)}&langpair=${fromLang}|${toLang}`;

  outputText.value = "Translating... ⏳";

  fetch(url)
    .then(response => response.json())
    .then(data => {
      const translated = data.responseData.translatedText;
      outputText.value = translated;
    })
    .catch(() => {
      outputText.value = "Error: Could not translate text.";
    });
});
