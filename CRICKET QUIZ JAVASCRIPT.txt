const quizData = [
    {
        question: "Who won the ICC Cricket World Cup in 2019?",
        choices: ["Australia", "New Zealand", "England", "India"],
        answer: 2
    },
    {
        question: "Which bowler has the most wickets in Test cricket?",
        choices: ["Shane Warne", "Muttiah Muralitharan", "James Anderson", "Anil Kumble"],
        answer: 1
    },
    {
        question: "Who scored the fastest century in ODI cricket?",
        choices: ["AB de Villiers", "Virat Kohli", "Chris Gayle", "Rohit Sharma"],
        answer: 0
    },
    {
        question: "Which Indian cricketer is known as 'The Wall'?",
        choices: ["Sachin Tendulkar", "Rahul Dravid", "Sourav Ganguly", "VVS Laxman"],
        answer: 1
    },
    {
        question: "Where was the first ever Test match played?",
        choices: ["England", "Australia", "South Africa", "India"],
        answer: 0
    }
];

let currentQuestion = 0;
let score = 0;
let selectedChoice = null;

const questionEl = document.getElementById('question');
const choicesEl = document.getElementById('choices');
const nextBtn = document.getElementById('nextBtn');
const resultEl = document.getElementById('result');
const progressEl = document.getElementById('progress');
const pointsEl = document.getElementById('points');

function loadQuestion() {
    selectedChoice = null;
    questionEl.textContent = quizData[currentQuestion].question;
    choicesEl.innerHTML = '';
    resultEl.textContent = '';
    nextBtn.disabled = true;

    progressEl.textContent = `Question ${currentQuestion + 1} of ${quizData.length}`;
    pointsEl.textContent = `Score: ${score}`;

    quizData[currentQuestion].choices.forEach((choice, idx) => {
        const li = document.createElement('li');
        const btn = document.createElement('button');
        btn.textContent = choice;
        btn.className = 'choice-btn';
        btn.onclick = () => selectChoice(btn, idx);
        li.appendChild(btn);
        choicesEl.appendChild(li);
    });
}

function selectChoice(btn, idx) {
    Array.from(document.getElementsByClassName('choice-btn')).forEach(b => b.disabled = true);
    selectedChoice = idx;
    nextBtn.disabled = false;

    if (idx === quizData[currentQuestion].answer) {
        score++;
        btn.classList.add('correct');
        resultEl.textContent = 'Correct!';
        resultEl.style.color = '#4CAF50';
    } else {
        btn.classList.add('wrong');
        const correctBtn = Array.from(document.getElementsByClassName('choice-btn'))[quizData[currentQuestion].answer];
        correctBtn.classList.add('correct');
        resultEl.textContent = `Wrong! Correct answer: ${quizData[currentQuestion].choices[quizData[currentQuestion].answer]}`;
        resultEl.style.color = '#f44336';
    }
    pointsEl.textContent = `Score: ${score}`;
}

nextBtn.onclick = function() {
    currentQuestion++;
    if (currentQuestion < quizData.length) {
        loadQuestion();
    } else {
        showResult();
    }
};

function showResult() {
    questionEl.textContent = 'Quiz Completed!';
    choicesEl.innerHTML = '';
    nextBtn.style.display = 'none';
    progressEl.textContent = '';
    resultEl.innerHTML = `
        Your score: <strong>${score}</strong> / ${quizData.length}<br>
        Percentage: <strong>${((score / quizData.length) * 100).toFixed(2)}%</strong><br>
        <button onclick="restartQuiz()">Restart Quiz</button>
    `;
}

function restartQuiz() {
    currentQuestion = 0;
    score = 0;
    nextBtn.style.display = '';
    loadQuestion();
}

loadQuestion();
